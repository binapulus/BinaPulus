🔥 Firebase Güvenlik Kuralları
Firebase Console → Firestore → Rules sekmesine yapıştır:
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /anayonetici/{doc} {
      allow read, write: if false;
    }
    match /binalar/{binaId}/veri/data {
      allow read, write: if request.time != null;
    }
    match /binalar/{binaId}/bildirimler/{doc} {
      allow read, write: if request.time != null;
    }
  }
}
